literal: "all" | "all!" | "infinity"

special-literal: STRING //provided at runtime can be mapped to any function that returns a number

function: NAME "(" expr(","expr)* ")"

atom: number | function | literal | special-literal

right-unary-op: % | # | "k" | "m" | "b" | "t"
left-unary-op: #

factor: atom | LPAREN expr RPAREN

mutate-expr: left-unary-op? factor right-unary-op?

higher-order-term: mutate-expr ((POW) mutate-expr)*

term: higher-order-term ((MUL | DIV) higher-order-term)*

arith-expr : term ((PLUS| MINUS) term)*

expr: arith-expr | literal

program: expr (SEMI expr)*
